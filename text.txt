fn help (x, z) {
	print 3;

	if x < z {
		return x;
	} else {
		return z;
	}
	return 0;
}

fn mult(x, y) {
	let mult_res = x * y;

	return mult_res;
}

fn fib(n) {
	if n == 1 {
		return 1;
	}
	if n == 0 {
		return 0;
	}

	return fib(n-1) + fib(n-2);
}

fn main() {
	let cond: bool;
	let i: int = 1;
	let x = 1;
	cond = i <= 10;
	while cond {
		print fib(i);

		i = i + 1;
		x = mult(x, i);
		print x;
		cond = i <= 10;
	}

	return 0;
}
